\begin{algorithm}[t]
\caption{Dynamics-aware set-cover CW inspection planner}
\label{alg:offline-planner}
\begin{algorithmic}[1]
\Require Station mesh $\mathcal{S}$, initial state $x_0$, camera model,
$d_{\mathrm{safe}}$, $C_{\mathrm{req}}$, $C_{\mathrm{stop}}$
\Ensure Planned trajectory $\mathcal{P}$ and inspection metrics
\State Sample area-weighted mesh targets
$\mathcal{T}=\{(p_i,n_i,A_i)\}_{i=1}^N$
\State Generate candidate viewpoints $\mathcal{V}=\{z_j\}_{j=1}^M$
\State Reject unsafe candidates and compute visibility matrix
$A\in\{0,1\}^{N\times M}$
\State Initialize covered set $\mathcal{C}\gets\emptyset$, selected list
$\mathcal{V}^{\star}\gets\emptyset$, and predicted state $x\gets x_0$
\While{$C(\mathcal{C})<C_{\mathrm{stop}}$ and the viewpoint budget is not exhausted}
  \State Roll out candidate CW transfers from $x$ and compute
  $\Delta A_j$ and $J_{\mathrm{CW}}(x,z_j)$
  \State Select $z^\star=\arg\max_j
  \Delta A_j/(1+J_{\mathrm{CW}}(x,z_j))+w_m d_{\min}$
  \State Append $z^\star$ to $\mathcal{V}^{\star}$
  \State Update $x$ using the terminal state of the transfer
  \State Update covered set $\mathcal{C}$ from the candidate visibility vector
\EndWhile
\State Reorder $\mathcal{V}^{\star}$ greedily using CW-transfer dynamic costs
\State Simulate ordered trajectory and compute coverage, attitude, and safety
\State Save trajectory, selected viewpoints, attitudes, visibility, and metrics
\State \Return $\mathcal{P}$
\end{algorithmic}
\end{algorithm}
